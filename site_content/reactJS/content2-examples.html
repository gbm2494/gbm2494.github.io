Example 2

  1. Inside project folder, create file calculator.jsx with this code:

    2. class Calculator {
       var num1 = 0;
       var num2 = 0;
       function constructor(x : number, y : number) {
         this.set(x, y);
       }
       function constructor(other : Calculator) {
         this.set(other);
       }
       function set(x : number, y : number) : void {
         this.num1 = x;
         this.num2 = y;
       }
       function getNum1() : number {
         return this.num1;
       }
       function getNum2() : number {
         return this.num2;
       }
       function set(other : Calculator) : void {
         this.set(other.num1, other.num2);
       }
       function sum() : number{
         return this.num1 + this.num2;
       }
    }
  2. Insert in same file this code:

    3. class _Main {
       static function main(args : string[]) : void {
         var calc = new Calculator(5, 10);
         log("Calculator with values num1 : " + calc.getNum1() + " , num2 : " + calc.getNum2());
         log("Sum result: ", calc.sum());
       }
    }
  3. Run example:

Example 3

  1. Inside project folder, create file example.jsx with this code:

    2. class Variants{
       function guessType(v : variant) : void {
        if (typeof v == "string") {
          log "param is string and the value is: " + v as string;
        }
        else {
          log "param is not string";
        }
       }
    }
  2. Insert in same file this code:

    3. class _Main {
       static function main(args : string[]) : void {
        //test int type
        var n : int = 5;
        log("int value : " + n);

        //test string type
        var s : string;
        s = "Gaudy";
        log("String value : " + s);

        //test array object
        var a = new Array.();
        a[0] = "value1";
        a[1] = "value2";
        a[2] = "value3";
        a.forEach(function(e){
         log("Position in vector : " + e);
        });

        //test map object
        var m = new Map.();
        m = {"key": 5};
        log("Map value with key : " + m["key"]);

        //test variant type
        var variant = new Variants();
        log("Sending int n as param...");
        variant.guessType(n);
        log("Sending string s as param...");
        variant.guessType(s);
        log("Sending int n as param, casting to string with expression 'as'...");
        variant.guessType(n as string);

        //test nullable type
        var s2 : Nullable.;
        s2 = "Blanco";
        log(s2);
        s2 = null;
        log(s2);
       }
    }
  3. Run example:

Examples applied to ReactJS

  • JSX is used with React to describe what the UI should look like, because it produces React elements.
  • Some key functionalities of JSX are mostly used on React:
    • Embedding JavaScripts expressions
    • Specifying attributes
    • Specifying children
    • Prevent injection attacks
    • JSX represents objects

ReactJS Example 1

Embedding JavaScripts expressions

const user = {
   firstName: 'Gaudy',
   lastName: 'Blanco'
};

const element = (
   <h1>
   Hello, {user.firstName}!
   </h1> );

ReactDOM.render(
   element,
   document.getElementById('root')
);

ReactJS Example 2

Specifying attributes

You can set html attributes using JSX values.

const user = {
   firstName: 'Gaudy',
   lastName: 'Blanco'
};

const element = (
   <h1>
   Hello, {user.firstName}!
   </h1> );

ReactDOM.render(
   element,
   document.getElementById('root')
); const user = {
   name = 'Gaudy',
   avatarUrl = '/img/photo1.img'
}

const element = <img src={user.avatarUrl}></img>;

ReactJS Example 3

Specifying children

  • Some JS keywords aren't allowed on JSX html tags:
    • class -> className
    • tabindex -> tabIndex
    • ReactDOM uses camelCase for these conflicts.
const element = (
   <div className="container">
    <h1>Hello!</h1>
    <h2>Good to see you here.</h2>
   </div>
);

ReactJS Example 4

Prevent injection attacks

  • Embed user input on JSX, ensures you never inject anything that's not explicity written on your application.
const title = response.potentiallyMaliciousInput;
// This is safe:
const element = <h1>{title}</h1>;

ReactJS Example 5

JSX represents objects

  • Embed user input on JSX, ensures you never inject anything that's not explicity written on your application.

JSX Syntax
const element = (
   <h1 className="greeting">
   Hello, world!
   </h1>
);
Pure JS Syntax
const element = React.createElement(
   'h1',
   {className: 'greeting'},
   'Hello, world!'
);

These codes produce the same object:

// Note: this structure is simplified
const element = {
   type: 'h1',
   props: {
      className: 'greeting',
      children: 'Hello, world'
   }
};
